### 1.5 日誌管理

```bash
#!/bin/bash
# 日誌輪替腳本 - 每週日執行

LOG_DIR="$HOME/.gemini/antigravity/scratch/financial_assistant/logs"
ARCHIVE_DIR="$LOG_DIR/archive"
DATE=$(date +%Y%m%d)

# 建立歸檔目錄
mkdir -p "$ARCHIVE_DIR"

# 壓縮並歸檔舊日誌
if [ -f "$LOG_DIR/stdout.log" ]; then
    gzip -c "$LOG_DIR/stdout.log" > "$ARCHIVE_DIR/stdout_$DATE.log.gz"
    > "$LOG_DIR/stdout.log"  # 清空檔案
fi

if [ -f "$LOG_DIR/stderr.log" ]; then
    gzip -c "$LOG_DIR/stderr.log" > "$ARCHIVE_DIR/stderr_$DATE.log.gz"
    > "$LOG_DIR/stderr.log"
fi

# 刪除 90 天前的舊日誌
find "$ARCHIVE_DIR" -name "*.log.gz" -mtime +90 -delete

echo "[$(date)] 日誌輪替完成"
```

```python
#!/usr/bin/env python3
"""Notion 儀表板統計數字自動更新"""

import os
import sys
from datetime import datetime, timedelta
from dotenv import load_dotenv
import requests

# 載入環境變數
load_dotenv()

NOTION_TOKEN = os.getenv("NOTION_TOKEN")
NOTION_VERSION = "2022-06-28"
DASHBOARD_PAGE_ID = os.getenv("NOTION_DASHBOARD_PAGE_ID")  # 需要在 .env 新增
DATABASE_ID = os.getenv("NOTION_DATABASE_ID")

HEADERS = {
    "Authorization": f"Bearer {NOTION_TOKEN}",
    "Notion-Version": NOTION_VERSION,
    "Content-Type": "application/json"
}

def query_bills_statistics():
    """查詢帳單統計資料"""
    url = f"https://api.notion.com/v1/databases/{DATABASE_ID}/query"
    
    # 查詢所有帳單
    all_bills = []
    start_cursor = None
    
    while True:
        payload = {"page_size": 100}
        if start_cursor:
            payload["start_cursor"] = start_cursor
        
        response = requests.post(url, headers=HEADERS, json=payload)
        if response.status_code != 200:
            print(f"❌ 查詢失敗: {response.text}")
            return None
        
        data = response.json()
        all_bills.extend(data.get("results", []))
        
        if not data.get("has_more"):
            break
        start_cursor = data.get("next_cursor")
    
    # 統計數據
    stats = {
        "total_count": len(all_bills),
        "pending_count": 0,
        "paid_count": 0,
        "overdue_count": 0,
        "total_amount": 0,
        "pending_amount": 0,
        "paid_amount": 0,
        "overdue_amount": 0,
        "current_month_count": 0,
        "current_month_amount": 0,
    }
    
    now = datetime.now()
    current_month = now.strftime("%Y-%m")
    
    for bill in all_bills:
        props = bill.get("properties", {})
        
        # 提取資料
        amount = props.get("金額", {}).get("number", 0) or 0
        status_obj = props.get("狀態", {}).get("select")
        status = status_obj.get("name") if status_obj else "未設定"
        
        due_date_obj = props.get("日期", {}).get("date")
        due_date = due_date_obj.get("start") if due_date_obj else None
        
        # 提取帳單月份（從 Name）
        title_list = props.get("Name", {}).get("title", [])
        full_name = "".join([t["plain_text"] for t in title_list])
        billing_month = full_name[:7] if full_name and len(full_name) >= 7 else None
        
        # 累計總金額
        stats["total_amount"] += amount
        
        # 判斷狀態
        if "已繳" in status or "已付" in status:
            stats["paid_count"] += 1
            stats["paid_amount"] += amount
        elif "待繳" in status or "未繳" in status:
            stats["pending_count"] += 1
            stats["pending_amount"] += amount
            
            # 判斷是否逾期
            if due_date:
                try:
                    due_dt = datetime.fromisoformat(due_date)
                    if due_dt < now:
                        stats["overdue_count"] += 1
                        stats["overdue_amount"] += amount
                except:
                    pass
        
        # 當月帳單統計
        if billing_month == current_month:
            stats["current_month_count"] += 1
            stats["current_month_amount"] += amount
    
    return stats

def update_dashboard_page(stats):
    """更新儀表板頁面的統計數字"""
    url = f"https://api.notion.com/v1/pages/{DASHBOARD_PAGE_ID}"
    
    # 更新頁面內容（使用 Callout 或 Database Properties）
    # 這裡假設儀表板使用頁面屬性儲存統計數字
    payload = {
        "properties": {
            "總帳單數": {"number": stats["total_count"]},
            "待繳數量": {"number": stats["pending_count"]},
            "已繳數量": {"number": stats["paid_count"]},
            "逾期數量": {"number": stats["overdue_count"]},
            "總金額": {"number": stats["total_amount"]},
            "待繳金額": {"number": stats["pending_amount"]},
            "已繳金額": {"number": stats["paid_amount"]},
            "本月帳單數": {"number": stats["current_month_count"]},
            "本月總額": {"number": stats["current_month_amount"]},
        }
    }
    
    response = requests.patch(url, headers=HEADERS, json=payload)
    
    if response.status_code == 200:
        print("✅ 儀表板統計數字已更新")
        return True
    else:
        print(f"❌ 更新失敗: {response.text}")
        return False

def main():
    """主程式"""
    print(f"🔄 開始更新儀表板統計數字... ({datetime.now().strftime('%Y-%m-%d %H:%M:%S')})")
    
    # 查詢統計
    stats = query_bills_statistics()
    if not stats:
        print("❌ 統計查詢失敗")
        sys.exit(1)
    
    # 顯示統計
    print("\n📊 統計結果：")
    print(f"  總帳單數: {stats['total_count']}")
    print(f"  待繳: {stats['pending_count']} 筆 (${stats['pending_amount']:,.0f})")
    print(f"  已繳: {stats['paid_count']} 筆 (${stats['paid_amount']:,.0f})")
    print(f"  逾期: {stats['overdue_count']} 筆 (${stats['overdue_amount']:,.0f})")
    print(f"  本月: {stats['current_month_count']} 筆 (${stats['current_month_amount']:,.0f})")
    
    # 更新儀表板
    # update_dashboard_page(stats)  # 需要先在 Notion 建立對應頁面屬性
    
    print("✅ 完成")

if __name__ == "__main__":
    main()
```

```python
def mark_bill_as_paid(bill_id):
    """標記帳單為已繳"""

            }
        }

    return response.status_code == 200

def auto_mark_paid_bills():
    """自動標記已繳帳單（根據繳款日期）"""
    # 查詢所有待繳帳單
    url = f"https://api.notion.com/v1/databases/{DATABASE_ID}/query"
    
    payload = {
        "filter": {
            "property": "狀態",
            "select": {
                "equals": "待繳"
            }
        }
    }
    
    response = requests.post(url, headers=HEADERS, json=payload)
    bills = response.json().get("results", [])
    
    marked_count = 0
    
    for bill in bills:
        props = bill.get("properties", {})
        
        # 檢查是否有「繳款日期」欄位
        payment_date_obj = props.get("繳款日期", {}).get("date")

                print(f"✅ 已標記: {bill['id']}")
```

### 2.3 逾期提醒

**腳本位置：** `scripts/check_overdue.py`

```python
#!/usr/bin/env python3
"""檢查逾期帳單並發送警告"""

import os
import sys
from datetime import datetime
from dotenv import load_dotenv
import requests

sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
from modules.telegram_notifier import send_telegram_alert

load_dotenv()

NOTION_TOKEN = os.getenv("NOTION_TOKEN")
DATABASE_ID = os.getenv("NOTION_DATABASE_ID")
NOTION_VERSION = "2022-06-28"

HEADERS = {
    "Authorization": f"Bearer {NOTION_TOKEN}",
    "Notion-Version": NOTION_VERSION,
    "Content-Type": "application/json"
}

def check_overdue_bills():
    """檢查逾期帳單"""
    url = f"https://api.notion.com/v1/databases/{DATABASE_ID}/query"
    
    # 查詢待繳帳單
    payload = {
        "filter": {
            "or": [
                {"property": "狀態", "select": {"equals": "待繳"}},
                {"property": "狀態", "select": {"equals": "未繳"}}
            ]
        }
    }
    
    response = requests.post(url, headers=HEADERS, json=payload)
    bills = response.json().get("results", [])
    
    now = datetime.now()
    overdue_bills = []
    
    for bill in bills:
        props = bill.get("properties", {})
        
        due_date_obj = props.get("日期", {}).get("date")
        if not due_date_obj:
            continue
        
        due_date_str = due_date_obj.get("start")
        if not due_date_str:
            continue
        
        try:
            due_dt = datetime.fromisoformat(due_date_str)
            if due_dt < now:
                # 提取資料
                title_list = props.get("Name", {}).get("title", [])
                name = "".join([t["plain_text"] for t in title_list])
                amount = props.get("金額", {}).get("number", 0)
                days_overdue = (now - due_dt).days
                
                overdue_bills.append({
                    "name": name,
                    "amount": amount,
                    "due_date": due_date_str,
                    "days_overdue": days_overdue
                })
        except:
            pass
    
    return overdue_bills

def main():
    """主程式"""
    print(f"🔍 檢查逾期帳單... ({datetime.now().strftime('%Y-%m-%d %H:%M:%S')})")
    
    overdue_bills = check_overdue_bills()
    
    if overdue_bills:
        print(f"⚠️ 發現 {len(overdue_bills)} 筆逾期帳單！")
        
        # 建立警告訊息
        message = f"🚨 *逾期帳單警告* 🚨\n\n"
        message += f"發現 {len(overdue_bills)} 筆逾期帳單：\n\n"
        
        for bill in overdue_bills:
            message += f"• *{bill['name']}*\n"
            message += f"  金額: ${bill['amount']:,.0f}\n"
            message += f"  截止日: {bill['due_date']}\n"
            message += f"  逾期: {bill['days_overdue']} 天\n\n"
        
        message += "請儘速繳款！"
        
        # 發送 Telegram 通知
        send_telegram_alert(message, urgent=True)
        
        print("✅ 已發送逾期警告")
    else:
        print("✅ 無逾期帳單")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""Telegram 通知模組"""

import os
import requests
from dotenv import load_dotenv

load_dotenv()

TELEGRAM_BOT_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN")
TELEGRAM_CHAT_ID = os.getenv("TELEGRAM_CHAT_ID")

def send_telegram_message(message, parse_mode="Markdown", urgent=False):
    """發送 Telegram 訊息"""
    if not TELEGRAM_BOT_TOKEN or not TELEGRAM_CHAT_ID:
        print("⚠️ 未設定 Telegram 通知")
        return False
    
    url = f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage"
    
    payload = {
        "chat_id": TELEGRAM_CHAT_ID,
        "text": message,
        "parse_mode": parse_mode,
        "disable_notification": not urgent  # 緊急訊息會發出通知音
    }
    
    try:
        response = requests.post(url, json=payload, timeout=10)
        if response.status_code == 200:
            print("✅ Telegram 訊息已發送")
            return True
        else:
            print(f"❌ Telegram 發送失敗: {response.text}")
            return False
    except Exception as e:
        print(f"❌ Telegram 發送錯誤: {e}")
        return False

def send_telegram_alert(message, urgent=True):
    """發送警告訊息（快捷方法）"""
    return send_telegram_message(message, urgent=urgent)

def notify_new_bill(bill_data):

    message += f"🏦 銀行: {bill_data.get('bank', '未知')}\n"

    message += f"📅 截止日: {bill_data.get('due_date', '未設定')}\n"
    message += f"📊 帳單月份: {bill_data.get('billing_month', '未設定')}\n\n"
    message += "已自動匯入至 Notion 儀表板 ✅"
    
    return send_telegram_message(message, urgent=False)

def notify_payment_reminder(bills_to_remind):
```

```python
    bills = response.json().get("results", [])
    
    now = datetime.now()
    bills_to_remind = []
    
    for bill in bills:
        props = bill.get("properties", {})
        
        due_date_obj = props.get("日期", {}).get("date")
        if not due_date_obj:
            continue
        
        due_date_str = due_date_obj.get("start")
        if not due_date_str:
            continue
        
        try:
            due_dt = datetime.fromisoformat(due_date_str)
            days_remaining = (due_dt - now).days
            
            # 檢查是否在提醒日範圍內
            if days_remaining in REMINDER_DAYS:
                title_list = props.get("Name", {}).get("title", [])
                name = "".join([t["plain_text"] for t in title_list])
                amount = props.get("金額", {}).get("number", 0)
                
                bills_to_remind.append({
                    "name": name,
                    "amount": amount,
                    "due_date": due_date_str,
                    "days_remaining": days_remaining
                })
        except:
            pass
    
    return bills_to_remind

def main():
    """主程式"""
    print(f"⏰ 檢查繳款提醒... ({datetime.now().strftime('%Y-%m-%d %H:%M:%S')})")
    
    bills = get_bills_to_remind()
    
    if bills:
        print(f"📢 需要提醒 {len(bills)} 筆帳單")
        notify_payment_reminder(bills)
    else:
        print("✅ 今日無需提醒")

if __name__ == "__main__":
    main()
```

```python
"""歸檔舊帳單（超過 6 個月的已繳帳單）"""

import os
import sys

from dotenv import load_dotenv
import requests

load_dotenv()

NOTION_TOKEN = os.getenv("NOTION_TOKEN")
DATABASE_ID = os.getenv("NOTION_DATABASE_ID")

def get_old_paid_bills():
    """取得 6 個月前的已繳帳單"""
    url = f"https://api.notion.com/v1/databases/{DATABASE_ID}/query"
    
    six_months_ago = (datetime.now() - timedelta(days=180)).strftime("%Y-%m-%d")
    
    payload = {
        "filter": {
            "and": [
                {
                    "property": "狀態",
                    "select": {"equals": "已繳"}
                },
                {
                    "property": "日期",
                    "date": {"before": six_months_ago}
                }
            ]
        }
    }
    
    response = requests.post(url, headers=HEADERS, json=payload)
    return response.json().get("results", [])

def move_to_archive(bill_id):
    """移動帳單到歸檔資料庫"""
    # 方法 1: 更新 parent（移到歸檔資料庫）
    url = f"https://api.notion.com/v1/pages/{bill_id}"
    
    payload = {
        "parent": {
            "database_id": ARCHIVE_DATABASE_ID
        }
    }
    
    response = requests.patch(url, headers=HEADERS, json=payload)
    return response.status_code == 200

def archive_old_bills():
    """執行歸檔"""
    print("📦 開始歸檔舊帳單...")
    
    old_bills = get_old_paid_bills()
    archived_count = 0
    
    for bill in old_bills:
        if move_to_archive(bill["id"]):
            archived_count += 1
            
            props = bill.get("properties", {})
            title_list = props.get("Name", {}).get("title", [])
            name = "".join([t["plain_text"] for t in title_list])
            print(f"✅ 已歸檔: {name}")
    
    print(f"✅ 共歸檔 {archived_count} 筆帳單")
    return archived_count

def main():
    if not ARCHIVE_DATABASE_ID:
        print("⚠️ 尚未設定 NOTION_ARCHIVE_DATABASE_ID")
        print("請先在 Notion 建立歸檔資料庫，並將 ID 加入 .env")

    archive_old_bills()

if __name__ == "__main__":
    main()
```

### 4.2 重複資料檢測

**腳本位置：** `scripts/check_duplicates.py`

```python
#!/usr/bin/env python3
"""檢測重複的帳單資料"""

import os
import sys
from datetime import datetime
from dotenv import load_dotenv
import requests
from collections import defaultdict

sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

load_dotenv()

NOTION_TOKEN = os.getenv("NOTION_TOKEN")
DATABASE_ID = os.getenv("NOTION_DATABASE_ID")
NOTION_VERSION = "2022-06-28"

HEADERS = {
    "Authorization": f"Bearer {NOTION_TOKEN}",
    "Notion-Version": NOTION_VERSION,
    "Content-Type": "application/json"
}

def get_all_bills():
    """取得所有帳單"""
    url = f"https://api.notion.com/v1/databases/{DATABASE_ID}/query"
    
    all_bills = []
    start_cursor = None
    
    while True:
        payload = {"page_size": 100}
        if start_cursor:
            payload["start_cursor"] = start_cursor
        
        response = requests.post(url, headers=HEADERS, json=payload)
        data = response.json()
        all_bills.extend(data.get("results", []))
        
        if not data.get("has_more"):
            break
        start_cursor = data.get("next_cursor")
    
    return all_bills

def extract_key(bill):
    """提取帳單唯一鍵（銀行 + 帳單月份 + 金額）"""
    props = bill.get("properties", {})
    
    # 銀行
    bank_obj = props.get("銀行名稱", {}).get("select")
    bank = bank_obj.get("name") if bank_obj else "未知"
    
    # 帳單月份（從 Name）
    title_list = props.get("Name", {}).get("title", [])
    full_name = "".join([t["plain_text"] for t in title_list])
    billing_month = full_name[:7] if full_name and len(full_name) >= 7 else "未知"
    
    # 金額
    amount = props.get("金額", {}).get("number", 0)
    
    return f"{bank}_{billing_month}_{amount}"

def find_duplicates():
    """尋找重複帳單"""
    print("🔍 檢查重複帳單...")
    
    bills = get_all_bills()
    bill_map = defaultdict(list)
    
    # 建立索引
    for bill in bills:
        key = extract_key(bill)
        bill_map[key].append(bill)
    
    # 找出重複
    duplicates = {k: v for k, v in bill_map.items() if len(v) > 1}
    
    if duplicates:
        print(f"⚠️ 發現 {len(duplicates)} 組重複帳單：\n")
        
        for key, bills_group in duplicates.items():
            print(f"重複組: {key}")
            for bill in bills_group:
                props = bill.get("properties", {})
                title_list = props.get("Name", {}).get("title", [])
                name = "".join([t["plain_text"] for t in title_list])
                created = bill.get("created_time", "")
                print(f"  - {name} (建立於 {created})")
                print(f"    ID: {bill['id']}")
            print()
        
        return duplicates
    else:
        print("✅ 無重複帳單")
        return {}

def main():
    duplicates = find_duplicates()
    
    if duplicates:
        print("\n💡 建議：")
        print("1. 手動檢查並刪除重複項目")
        print("2. 或執行 scripts/remove_duplicates.py 自動刪除（保留最新）")

if __name__ == "__main__":
    main()
```

### 5.2 月度報告自動生成

**腳本位置：** `scripts/generate_monthly_report.py`

```python
#!/usr/bin/env python3
"""生成月度報告並發送至 Telegram"""

import os
import sys
from datetime import datetime
from dotenv import load_dotenv
import requests

sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
from modules.telegram_notifier import send_telegram_message

load_dotenv()

NOTION_TOKEN = os.getenv("NOTION_TOKEN")
DATABASE_ID = os.getenv("NOTION_DATABASE_ID")
NOTION_VERSION = "2022-06-28"

HEADERS = {
    "Authorization": f"Bearer {NOTION_TOKEN}",
    "Notion-Version": NOTION_VERSION,
    "Content-Type": "application/json"
}

def generate_monthly_report(year, month):
    """生成指定月份的報告"""
    url = f"https://api.notion.com/v1/databases/{DATABASE_ID}/query"
    
    # 查詢該月份的所有帳單
    target_month = f"{year}-{month:02d}"
    
    response = requests.post(url, headers=HEADERS, json={})
    all_bills = response.json().get("results", [])
    
    # 篩選該月份
    monthly_bills = []
    for bill in all_bills:
        props = bill.get("properties", {})
        title_list = props.get("Name", {}).get("title", [])
        full_name = "".join([t["plain_text"] for t in title_list])
        
        if full_name.startswith(target_month):
            amount = props.get("金額", {}).get("number", 0)
            status_obj = props.get("狀態", {}).get("select")
            status = status_obj.get("name") if status_obj else "未設定"
            bank_obj = props.get("銀行名稱", {}).get("select")
            bank = bank_obj.get("name") if bank_obj else "未知"
            
            monthly_bills.append({
                "bank": bank,
                "amount": amount,
                "status": status
            })
    
    # 統計
    total_count = len(monthly_bills)
    total_amount = sum(b["amount"] for b in monthly_bills)
    paid_count = sum(1 for b in monthly_bills if "已繳" in b["status"])
    paid_amount = sum(b["amount"] for b in monthly_bills if "已繳" in b["status"])
    pending_count = total_count - paid_count
    pending_amount = total_amount - paid_amount
    
    # 按銀行統計
    bank_stats = {}
    for bill in monthly_bills:
        bank = bill["bank"]
        if bank not in bank_stats:
            bank_stats[bank] = {"count": 0, "amount": 0}
        bank_stats[bank]["count"] += 1
        bank_stats[bank]["amount"] += bill["amount"]
    
    # 生成報告
    report = f"📊 *{year} 年 {month} 月信用卡帳單報告*\n\n"
    report += f"📋 總覽：\n"
    report += f"  • 總帳單數：{total_count} 筆\n"
    report += f"  • 總金額：${total_amount:,.0f}\n"
    report += f"  • 已繳：{paid_count} 筆 (${paid_amount:,.0f})\n"
    report += f"  • 待繳：{pending_count} 筆 (${pending_amount:,.0f})\n\n"
    
    report += f"🏦 按銀行統計：\n"
    for bank, stats in sorted(bank_stats.items(), key=lambda x: x[1]["amount"], reverse=True):
        report += f"  • {bank}：{stats['count']} 筆, ${stats['amount']:,.0f}\n"
    
    report += f"\n📅 報告生成時間：{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}"
    
    return report

def main():
    """主程式"""
    # 預設生成上個月的報告
    now = datetime.now()
    if now.month == 1:
        year = now.year - 1
        month = 12
    else:
        year = now.year
        month = now.month - 1
    
    # 也可以從命令列參數指定
    if len(sys.argv) == 3:
        year = int(sys.argv[1])
        month = int(sys.argv[2])
    
    print(f"📊 生成 {year}-{month:02d} 月度報告...")
    
    report = generate_monthly_report(year, month)
    print(report)
    
    # 發送至 Telegram
    send_telegram_message(report, urgent=False)
    print("✅ 報告已發送至 Telegram")

if __name__ == "__main__":
    main()
```
